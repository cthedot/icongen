icongen - Web and App icons and PNG favicons

Generate Web and App Icons from a single source image locally in your browser. Ideally use an SVG image.

browserconfig.xml for IE (Windows 8.1) and Edge

<?xml version="1.0" encoding="utf-8" ?>
<browserconfig>
  <msapplication>
    <tile>
    <square70x70logo src="/iconx/browserconfig/tile70x70.png" />
    <square150x150logo src="/iconx/browserconfig/tile150x150.png" />
    <wide310x150logo src="/iconx/browserconfig/tile310x150.png" />
    <square310x310logo src="/iconx/browserconfig/tile310x310.png" />
    <tilecolor>#550000</tilecolor>
    </tile>
  </msapplication>
</browserconfig>
